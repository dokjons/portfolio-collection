                    class Education:
                        def __init__(self, name, school, country, start_year, finish_year):
                            self.name = name
                            self.school = school
                            self.country = country
                            self.start_year = start_year
                            self.finish_year = finish_year
                    
                    class Person:
                        def __init__(self, name, email, skills, education, experience):
                            self.name = name
                            self.email = email
                            self.skills = skills
                            self.education = education
                            self.experience = experience
                    
                    def create_person(name, email, skills, education, experience):
                        return Person(name, email, skills, education, experience)
                    
                    # Create a person instance with my information
                    # Shows name, skills, education, and latest experince
                    person = create_person(
                        'Jonas Jensen',
                        'jonasjensen321@gmail.com',
                        ['Python', 'HTML5/CSS3/JS', 'SCSS/LESS', 'Powershell', 'Java', 'SQL/MySQL/MariaDB/LightSQL', 'Photoshop', 'Illustrator', 'Adobe xD/Figma'],
                        [
                            Education('Multimedia Designer', 'IBA Erhvervsakademi', 'Denmark', '2022', '2024'),
                            Education('IT Supporter', 'Syddansk Erhvervsskole', 'Denmark', '2016', '2019'),
                        ],
                        ['Amazon Web Services', 'Attech Computer ApS', 'LAKUDA']
                    )
                    
                    # Prints the information about me
                    print(f"Name: {person.name}")
                    print(f"Mail: {person.email}")
                    print("Skills:")
                    for skill in person.skills:
                        print(f"- {skill}")
                    print("Education:")
                    for edu in person.education:
                        print(f"- {edu.name}")
                        print(f"  School: {edu.school}")
                        print(f"  Location: {edu.country}")
                        print(f"  Years: {edu.start_year} - {edu.finish_year}")
                    print("Experience:")
                    for exp in person.experience:
                        print(f"- {exp}")
